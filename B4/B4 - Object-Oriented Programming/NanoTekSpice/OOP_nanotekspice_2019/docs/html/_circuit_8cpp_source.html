166 {
167  std::vector<std::unique_ptr<IComponent>> &circuit = this->getCircuit();

171  if (oc->getName() == linked1)
172  return true;
173  return false;
174  });

176  auto const& selectedComponentU2 = std::find_if(circuit.begin(), circuit.end(), [linked2](std::unique_ptr<IComponent> &o)

178  auto *oc = static_cast<nts::AComponent *>(o.get());
179  if (oc->getName() == linked2)
180  return true;
181  return false;
182  });

184  if (selectedComponentU1 == circuit.end() || selectedComponentU2 == circuit.end())
185  throw nts::FileError("Try to link component that do not exist", "File");
186  auto *selectedComponent1 = static_cast<nts::AComponent *>(selectedComponentU1->get());
187  auto *selectedComponent2 = static_cast<nts::AComponent *>(selectedComponentU2->get());
188  if (nts::Utility::isOutput(selectedComponent1) && nts::Utility::isOutput(selectedComponent2))
189  throw nts::FileError("Try to link two output together", "File");
190  if (nts::Utility::isInput(selectedComponent1) && nts::Utility::isInput(selectedComponent2))
191  throw nts::FileError("Try to link two input together", "File");
192  if (nts::Utility::isOutput(selectedComponent1))
193  selectedComponent2->setLink(pinLinked2, *selectedComponent1, pinLinked1);
194  else if (nts::Utility::isOutput(selectedComponent1))

247 void nts::Circuit::simulate()
248 {

250  size_t nb_circuit = 0;

252  std::for_each(_total.begin(), _total.end(), [&nb_circuit](std::unique_ptr<IComponent> &component) {
253  if (!nts::Utility::isInput(component.get()) && !nts::Utility::isOutput(component.get())) {

261  if (!nts::Utility::isInput(component.get()) && !nts::Utility::isOutput(component.get())) {
262  component.get()->run();
263  }
264  });
265  }
